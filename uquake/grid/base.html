 69    def __init__(self, data_or_dims: Union[np.ndarray, List, Tuple],
 70                 spacing: Union[np.ndarray, List, Tuple] = None,
 71                 origin: Union[np.ndarray, List, Tuple] = None,
 72                 resource_id: ResourceIdentifier = ResourceIdentifier(),
 73                 value: float = 0,
 74                 coordinate_system: CoordinateSystem = CoordinateSystem.NED,
 75                 label: str = __default_grid_label__):
 76
 77        """
 78        can hold both 2 and 3 dimensional grid
 79        :param data_or_dims: either a numpy array or a Tuple/List with the grid
 80        dimensions. If grid dimensions are specified, the grid is initialized
 81        with value
 82        :param spacing: a set of two or three values containing the grid spacing
 83        :type spacing: Tuple, List or numpy.ndarray
 84        :param origin: a set of two or three values origin of the grid
 85        :type origin: tuple, List or numpy.ndarray
 86        :param resource_id: unique identifier for the grid, if set to None,
 87        :param value: value to fill the grid should dims be specified
 88        :type value:
 89        uuid4 is used to define a unique identifier.
 90        :param coordinate_system: Coordinate system
 91        :type coordinate_system: ~uquake.core.coordinates.CoordinateSystem
 92        :param label: Label providing additional information on the grid usage
 93        :type label: str
 94        """
 95
 96        data_or_dims = np.array(data_or_dims)
 97
 98        if data_or_dims.ndim == 1:
 99            self.data = np.ones(data_or_dims) * value
100        else:
101            self.data = data_or_dims
102
103        if resource_id is None:
104            self.resource_id = str(uuid4())
105        else:
106            self.resource_id = resource_id
107
108        if origin is None:
109            self.origin = np.zeros(len(self.data.shape))
110        else:
111            origin = np.array(origin)
112            if origin.shape[0] == len(self.data.shape):
113                self.origin = origin
114            else:
115                logger.error(f'origin shape should be {len(self.data.shape)}')
116                raise ValueError
117
118        if spacing is None:
119            self.spacing = np.ones(len(self.data.shape))
120        else:
121            spacing = np.array(spacing)
122            if spacing.shape[0] == len(self.data.shape):
123                self.spacing = spacing
124            else:
125                logger.error(f'spacing shape should be {len(self.data.shape)}')
126                raise ValueError
127
128            self.coordinate_system = coordinate_system
129
130        self.id = id
131        self.label = label

148    def from_ods(cls, origin, dimensions, spacing, val=0):
149        """
150        create a grid from origin, dimensions and spacing
151        :param origin: grid origin
152        :type origin: tuple
153        :param dimensions: grid dimension
154        :type dimensions: tuple
155        :param spacing: spacing between the grid nodes
156        :type spacing: float
157        :param val: constant value with which to fill the grid
158        """
159
160        data = np.ones(tuple(dimensions)) * val
161        cls.grid = cls(data, spacing=spacing, origin=origin)

163    @classmethod
164    def from_ocs(cls, origin, corner, spacing, val=0):
165        """
166        create a grid from origin, corner and spacing
167        :param origin: grid origin (e.g., lower left corner for 2D grid)
168        :type origin: tuple or list or numpy.array
169        :param corner: grid upper (e.g., upper right corner for 2D grid)
170        :type corner: tuple or list or numpy.array
171        :param spacing: spacing between the grid nodes
172        :type spacing: float
173        :param val: constant value with which to fill the grid
174        :param buf: buffer around the grid in fraction of grid size
175        """
176        origin2 = origin
177        corner2 = corner
178
179        gshape = tuple([int(np.ceil((c - o) / s))
180                        for o, c, s in zip(origin2, corner2, spacing)])
181        data = np.ones(gshape) * val
182        out = cls(data, spacing=spacing, origin=origin)
183        out.fill_homogeneous(val)
184        return out

186    @classmethod
187    def from_ocd(cls, origin, corner, dimensions, val=0):
188        """
189        create a grid from origin, corner and dimensions
190        :param origin: grid origin (e.g., lower left corner for 2D grid)
191        :param corner: grid upper (e.g., upper right corner for 2D grid)
192        :param dimensions: grid dimensions
193        :param val: constant value with which to fill the grid
194        :return:
195        """
196
197        data = np.ones(dimensions) * val
198        spacing = (corner - origin) / (dimensions - 1)
199        cls(data, spacing, spacing=spacing, origin=origin)
200        return cls

299    def in_grid(self, point):
300        """
301        Check if a point is inside the grid
302        :param point: the point to check in absolute coordinate (model)
303        :type point: tuple, list or numpy array
304        :returns: True if point is inside the grid
305        :rtype: bool
306        """
307        corner1 = self.origin
308        corner2 = self.origin + self.spacing * np.array(self.shape)
309
310        return np.all((point >= corner1) & (point <= corner2))

319    def fill_random(self, mean, std, smooth_sigma, seed=None):
320        """
321        fill the model with random number with a mean of "mean".
322        :param mean: mean of the random number
323        :param std: standard deviation of the grid
324        :param sigma: gaussian smoothing parameter
325        :param seed: random seed
326        """

328        np.random.seed(seed)
329
330        self.data = np.random.randn(self.dims[0], self.dims[1], self.dims[2])
331        self.smooth(smooth_sigma)
332        self.data = self.data * std / np.std(self.data) + mean

334    def generate_points(self, pt_spacing=None):
335        """
336        Generate points within the grid
337        """
338        # if pt_spacing is None:
339        ev_spacing = self.spacing
340
341        dimensions = np.array(self.shape) * self.spacing / ev_spacing
342
343        xe = np.arange(0, dimensions[0]) * ev_spacing[0] + self.origin[0]
344        ye = np.arange(0, dimensions[1]) * ev_spacing[1] + self.origin[1]
345        ze = np.arange(0, dimensions[2]) * ev_spacing[2] + self.origin[2]
346
347        Xe, Ye, Ze = np.meshgrid(xe, ye, ze)
348
349        Xe = Xe.reshape(np.prod(Xe.shape))
350        Ye = Ye.reshape(np.prod(Ye.shape))
351        Ze = Ze.reshape(np.prod(Ze.shape))
352        data_e = self.data.reshape(np.prod(self.data.shape))
353        return Xe, Ye, Ze, data_e

358    def rbf_interpolation_sensitivity(self, location, epsilon,
359                                      threshold=0.1):
360        """
361        calculate the sensitivity of each element given a location
362        :param location: location in model space at which the interpolation
363        occurs
364        :param epsilon: the standard deviation of the gaussian kernel
365        :param threshold: threshold relative to the maximum value below which
366        the weights are considered 0.
367        :rparam: the sensitivity matrix
368        """
369        x, y, z, v = self.flattens()
370
371        # calculating the distance between the location and every grid points
372
373        dist = np.linalg.norm([x - location[0], y - location[1],
374                               z - location[2]], axis=0)
375
376        sensitivity = np.exp(-(dist / epsilon) ** 2)
377        sensitivity[sensitivity < np.max(sensitivity) * threshold] = 0
378        sensitivity = sensitivity / np.sum(sensitivity)
379
380        return sensitivity

382    def generate_random_points_in_grid(self, n_points=1,

385        """
386        Generate a random set of points within the grid
387        :param n_points: number of points to generate (default=1)
388        :type n_points: int
389        :param grid_space: whether the output is expressed in
390        grid coordinates (True) or model coordinates (False)

397        np.random.seed(seed)
398
399        points = np.random.uniform(0, 1, (n_points, len(self.data.shape)))

404        if not grid_space:
405            return self.transform_from_grid(points)
406
407        return points

409    def interpolate(self, coord, grid_space=True, mode='nearest',
410                    order=1, **kwargs):
411        """
412        This function interpolate the values at a given point expressed
413        either in grid or absolute coordinates
414        :param coord: Coordinate of the point(s) at which to interpolate
415        either in grid or absolute coordinates
416        :type coord: list, tuple, numpy.array
417        :param grid_space: true if the coordinates are expressed in
418        grid space (indices can be float) as opposed to model space
419        :param mode: {'reflect', 'constant', 'nearest', 'mirror', 'wrap'},
420        optional
421
422        The `mode` parameter determines how the input array is extended
423        beyond its boundaries. Default is 'constant'. Behavior for each valid
424        value is as follows:
425
426        'reflect' (`d c b a | a b c d | d c b a`)
427            The input is extended by reflecting about the edge of the last
428            pixel.
429
430        'constant' (`k k k k | a b c d | k k k k`)
431            The input is extended by filling all values beyond the edge with
432            the same constant value, defined by the `cval` parameter.
433
434        'nearest' (`a a a a | a b c d | d d d d`)
435            The input is extended by replicating the last pixel.
436
437        'mirror' (`d c b | a b c d | c b a`)
438            The input is extended by reflecting about the center of the last
439            pixel.
440
441        'wrap' (`a b c d | a b c d | a b c d`)
442            The input is extended by wrapping around to the opposite edge.
443
444        :param order: int, optional
445            The order of the spline interpolation, default is 3.
446            The order has to be in the range 0-5.
447        :type order: int
448
449        :type grid_space: bool
450        :rtype: numpy.array
451        """
452
453        coord = np.array(coord)
454
455        if not grid_space:
456            coord = self.transform_to(coord)
457
458        if len(coord.shape) < 2:
459            coord = coord[:, np.newaxis]
460
461        try:
462            return map_coordinates(self.data, coord, mode=mode, order=order,
463                                   **kwargs)
464        except Exception as e:
465            # logger.warning(e)
466            # logger.info('transposing the coordinate array')
467            return map_coordinates(self.data, coord.T, mode=mode, order=order,
468                                   **kwargs)

572    def smooth(self,
573               sigma: Union[np.ndarray, List[Union[int, float]], Union[int, float]],
574               grid_space: bool = False,
575               preserve_statistic: bool = True,
576               in_place: bool = True) -> Grid:
577        """
578        Smooth the data using a Gaussian filter.
579
580        :param sigma: Standard deviation for Gaussian filter.
581                      Can be a single number or a numpy array.
582        :param grid_space: If True, smoothing is performed in grid space.
583        :param preserve_statistic: If True, original mean and standard deviation are preserved.
584        :param in_place: If True, modifies the object's data in place. Otherwise, returns a new object.
585        :return: self if in_place is True, otherwise a new instance of Grid.
586        """
587
588        # Use self's properties or create a copy for a new object.
589        obj = self if in_place else deepcopy(
590            self)  # You need to import deepcopy from the copy module
591
592        original_mean = np.mean(obj.data)
593        original_std = np.std(obj.data)
594
595        if isinstance(sigma, (int, float)):
596            sigma = np.array([sigma] * obj.data.ndim)
597
598        if isinstance(sigma, list):
599            sigma = np.array(sigma)
600
601        if not grid_space:
602            sigma = sigma / obj.spacing
603
604        smoothed_data = gaussian_filter(obj.data, sigma)
605
606        if preserve_statistic:
607            smoothed_mean = np.mean(smoothed_data)
608            smoothed_std = np.std(smoothed_data)
609            obj.data = (smoothed_data - smoothed_mean) * (
610                        original_std / smoothed_std) + original_mean
611        else:
612            obj.data = smoothed_data
613
614        return obj

649    def generate_random_inventory_in_grid(self, num_station: int = 1,
650                                          ratio_uni_tri: float = 3) -> Inventory:
651        """
652        Generate a random inventory within the grid.
653

655        :type num_station: int

657        :type ratio_uni_tri: float

665        # Generate random points

667
668        # Calculate number of uniaxial and triaxial stations based on the ratio
669
670        for i, point in enumerate(points):
671            # Create coordinates and a Station
672            coordinates = Coordinates(point[0], point[1], point[2],
673                                      coordinate_system=self.coordinate_system)

675            station = Station(code=f"ST{i:02d}", coordinates=coordinates)

677            # Determine if this station should be uniaxial or triaxial
678            is_uni = random.choices([True, False], [ratio_uni_tri, 1])[0]

680            # Create Channel(s)

682                channel = Channel(code="HHZ", location_code="00",

684                station.channels.append(channel)
685            else:
686                for axis in ["Z", "N", "E"]:
687                    channel = Channel(code=f"HH{axis}", location_code=f"00",
688                                      coordinates=coordinates)
689                    station.channels.append(channel)

731def ray_tracer(travel_time_grid, start, grid_space=False, max_iter=1000,
732               arrival_id=None, earth_model_id=None,
733               network: str=None):
734    """
735    This function calculates the ray between a starting point (start) and an
736    end point, which should be the seed of the travel_time grid, using the
737    gradient descent method.
738    :param travel_time_grid: a travel time grid
739    :type travel_time_grid: TTGrid
740    :param start: the starting point (usually event location)
741    :type start: tuple, list or numpy.array
742    :param grid_space: true if the coordinates are expressed in
743    grid space (indices can be fractional) as opposed to model space
744    (x, y, z)
745    :param max_iter: maximum number of iteration
746    :param arrival_id: id of the arrival associated to the ray if
747    applicable
748    :type arrival_id: uquake.core.event.ResourceIdentifier
749    :param earth_model_id: velocity/earth model id.
750    :type earth_model_id: uquake.core.event.ResourceIdentifier
751    :param network: network information
752    :type network: str
753    :rtype: numpy.array
754    """
755
756    from uquake.core.event import Ray
757
758    interpolation_order = 1
759
760    if grid_space:
761        start = np.array(start)
762        start = travel_time_grid.transform_from(start)
763
764    origin = travel_time_grid.origin
765    spacing = travel_time_grid.spacing
766    end = np.array(travel_time_grid.seed)
767    start = np.array(start)
768
769    # calculating the gradient in every dimension at every grid points
770    gds = [Grid(gd, origin=origin, spacing=spacing)
771           for gd in np.gradient(travel_time_grid.data)]
772
773    dist = np.linalg.norm(start - end)
774    cloc = start  # initializing cloc "current location" to start
775    spacing = np.linalg.norm(spacing)
776    gamma = spacing / 2  # gamma is set to half the grid spacing. This
777    # should be
778    # sufficient. Note that gamma is fixed to reduce
779    # processing time.
780    nodes = [start]
781
782    iter_number = 0
783    while np.all(dist > spacing / 2):
784        if iter_number > max_iter:
785            break
786
787        # if dist < spacing * 4:
788        #     gamma = spacing / 4
789
790        gvect = np.array([gd.interpolate(cloc, grid_space=False,
791                                         order=interpolation_order)[0]
792                          for gd in gds])
793
794        if np.linalg.norm(gvect) == 0:
795            break
796
797        dr = gamma * gvect / (np.linalg.norm(gvect) + 1e-8)
798
799        if np.linalg.norm(dr) < gamma / 2:
800            dr = (dr / np.linalg.norm(dr)) * gamma / 2
801
802        cloc = cloc - dr
803
804        nodes.append(cloc)
805        dist = np.linalg.norm(cloc - end)
806
807        iter_number += 1
808
809    nodes.append(end)
810
811    tt = travel_time_grid.interpolate(start, grid_space=False,
812                                      order=interpolation_order)[0]
813
814    az = travel_time_grid.to_azimuth_point(start, grid_space=False,
815                                           order=interpolation_order)
816    toa = travel_time_grid.to_takeoff_point(start, grid_space=False,
817                                            order=interpolation_order)
818
819    ray = Ray(nodes=nodes, waveform_id=travel_time_grid.waveform_id,
820              arrival_id=arrival_id, phase=travel_time_grid.phase,
821              azimuth=az, takeoff_angle=toa, travel_time=tt,
822              earth_model_id=earth_model_id, network=network)
823
824    return ray
