 249    def __getMergeId(self, tr):
 250        tr_id = tr.id
 251        # don't merge normal traces with previews
 252        try:
 253            if tr.stats.preview:
 254                tr_id += 'preview'
 255        except AttributeError:
 256            pass
 257        # don't merge traces with different processing steps
 258        try:
 259            if tr.stats.processing:
 260                tr_id += str(tr.stats.processing)
 261        except AttributeError:
 262            pass
 263        return tr_id

 265    def __getMergablesIds(self):
 266        ids = set()

 271    def plotWaveform(self, *args, **kwargs):
 272        """
 273        Creates a graph of any given ObsPy Stream object. It either saves the
 274        image directly to the file system or returns a binary image string.
 275
 276        For all color values you can use legit HTML names, HTML hex strings
 277        (e.g. '#eeefff') or you can pass an RGB tuple, where each of R, G, and
 278        B are in the range [0, 1]. You can also use single letters for basic
 279        built-in colors ('b' = blue, 'g' = green, 'r' = red, 'c' = cyan,
 280        'm' = magenta, 'y' = yellow, 'k' = black, 'w' = white) and gray shades
 281        can be given as a string encoding a float in the 0-1 range.
 282        """
 283        # Setup the figure if not passed explicitly.
 284        if not self.fig_obj:
 285            self.__setupFigure()
 286        else:
 287            self.fig = self.fig_obj
 288        # Determine kind of plot and do the actual plotting.
 289        if self.type == 'dayplot':
 290            self.plotDay(*args, **kwargs)
 291        elif self.type == 'section':
 292            self.plotSection(*args, **kwargs)
 293        else:
 294            self.plot(*args, **kwargs)
 295
 296        # Adjust the subplot so there is always a fixed margin on every side
 297        if self.type != 'dayplot':
 298            fract_y = 60.0 / self.height
 299            fract_y2 = 40.0 / self.height
 300            fract_x = 80.0 / self.width
 301            self.fig.subplots_adjust(top=1.0 - fract_y, bottom=fract_y2,
 302                                     left=fract_x, right=1.0 - fract_x / 2)
 303        with warnings.catch_warnings(record=True):
 304            warnings.filterwarnings("ignore", DATELOCATOR_WARNING_MSG,
 305                                    UserWarning, "matplotlib.dates")
 306            if self.draw:
 307                self.fig.canvas.draw()
 308            # The following just serves as a unified way of saving and
 309            # displaying the plots.
 310            if not self.transparent:
 311                extra_args = {'dpi': self.dpi,
 312                              'facecolor': self.face_color,
 313                              'edgecolor': self.face_color}
 314            else:
 315                extra_args = {'dpi': self.dpi,
 316                              'transparent': self.transparent,
 317                              'facecolor': 'k'}
 318            if self.outfile:
 319                # If format is set use it.
 320                if self.format:
 321                    self.fig.savefig(self.outfile, format=self.format,
 322                                     **extra_args)
 323                # Otherwise use format from self.outfile or default to PNG.
 324                else:
 325                    self.fig.savefig(self.outfile, **extra_args)
 326            else:
 327                # Return a binary image string if not self.outfile but
 328                # self.format.
 329                if self.format:
 330                    imgdata = io.BytesIO()
 331                    self.fig.savefig(imgdata, format=self.format,
 332                                     **extra_args)
 333                    imgdata.seek(0)
 334                    return imgdata.read()
 335                elif self.handle:
 336                    return self.fig
 337                else:
 338                    if not self.fig_obj and self.show:
 339                        try:
 340                            plt.show(block=self.block)
 341                        except:
 342                            plt.show()

 344    def plot(self, *args, **kwargs):
 345        """
 346        Plot the Traces showing one graph per Trace.
 347
 348        Plots the whole time series for self.max_npts points and less. For more
 349        points it plots minmax values.
 350        """
 351        stream_new = []
 352        # Just remove empty traces.
 353        if not self.automerge:
 354            for tr in self.stream:
 355                stream_new.append([])
 356                if len(tr.data):
 357                    stream_new[-1].append(tr)
 358        else:
 359            # Generate sorted list of traces (no copy)
 360            # Sort order: id, starttime, endtime
 361            ids = self.__getMergablesIds()
 362            for id in ids:
 363                stream_new.append([])
 364                for tr in self.stream:
 365                    tr_id = self.__getMergeId(tr)
 366                    if tr_id == id:
 367                        # does not copy the elements of the data array
 368                        tr_ref = copy(tr)
 369                        if tr_ref.data.size:
 370                            stream_new[-1].append(tr_ref)
 371                # delete if empty list
 372                if not len(stream_new[-1]):
 373                    stream_new.pop()
 374                    continue
 375        # If everything is lost in the process raise an Exception.
 376        if not len(stream_new):
 377            raise Exception("Nothing to plot")
 378        # Create helper variable to track ids and min/max/mean values.
 379        self.ids = []
 380        # Loop over each Trace and call the appropriate plotting method.
 381        self.axis = []
 382
 383        import numpy as np
 384        if self.event and self.location:
 385            origin = self.event.preferred_origin()
 386            if origin:
 387                ex = origin.x
 388                ey = origin.y
 389                ez = origin.z
 390                eloc = np.array([ex, ey, ez])
 391
 392                traces = []
 393                name = []
 394                dist = []
 395
 396                for sta in self.location.stations():
 397                    sloc = np.array([sta.x, sta.y, sta.z])
 398                    dist.append(np.linalg.norm(eloc - sloc))
 399                    name.append(sta.code)
 400
 401                indices = np.argsort(dist)
 402                name = np.array(name)
 403                traces = []
 404                for nm in name[indices]:
 405                    for tr in stream_new:
 406                        if tr[0].stats.station == nm:
 407                            traces.append(tr)
 408
 409                stream_new = traces
 410
 411        for _i, tr in enumerate(stream_new):
 412            # Each trace needs to have the same sampling rate.
 413            sampling_rates = set([_tr.stats.sampling_rate for _tr in tr])
 414            if len(sampling_rates) > 1:
 415                msg = "All traces with the same id need to have the same " + \
 416                      "sampling rate."
 417                raise Exception(msg)
 418            sampling_rate = sampling_rates.pop()
 419            if _i == 0:
 420                sharex = None
 421            else:
 422                sharex = self.axis[0]
 423            ax = self.fig.add_subplot(len(stream_new), 1, _i + 1,
 424                                      # axisbg=self.background_color,
 425                                      sharex=sharex)
 426            self.axis.append(ax)
 427            # XXX: Also enable the minmax plotting for previews.
 428            method_ = self.plotting_method
 429            if method_ is None:
 430                if ((self.endtime - self.starttime) * sampling_rate >
 431                        self.max_npts):
 432                    method_ = "fast"
 433                else:
 434                    method_ = "full"
 435            method_ = method_.lower()
 436            if method_ == 'full':
 437                self.__plotStraight(stream_new[_i], ax, *args, **kwargs)
 438            elif method_ == 'fast':
 439                self.__plotMinMax(stream_new[_i], ax, *args, **kwargs)
 440            else:
 441                msg = "Invalid plot method: '%s'" % method_
 442                raise ValueError(msg)
 443
 444            # plot picks
 445            if self.event:
 446                colors = {'P': 'b', 'S': 'r'}
 447                for arrival in self.event.preferred_origin().arrivals:
 448                    pk = arrival.get_pick()
 449                    if (pk.waveform_id.station_code == tr[0].stats.station) \
 450                            & (pk.waveform_id.location_code ==
 451                               tr[0].stats.location):
 452                        time = date2num(pk.time.datetime)
 453                        ax.axvline(time, ls='-',
 454                                   c=colors[pk.phase_hint.upper()])
 455                        if arrival.time_residual is not None:
 456                            # predicted_time = time - arrival.time_residual
 457                            predicted_time = pk.time - \
 458                                             arrival.time_residual
 459                            ax.axvline(date2num(predicted_time),
 460                                       ls='--',
 461                                       c=colors[pk.phase_hint.upper()])
 462                            # raspoutine
 463
 464        # Set ticks.
 465        self.__plotSetXTicks()
 466        self.__plotSetYTicks()
 467        xmin = self._time_to_xvalue(self.starttime)
 468        xmax = self._time_to_xvalue(self.endtime)
 469        ax.set_xlim(xmin, xmax)
 470        self._draw_overlap_axvspan_legend()

 472    def plotDay(self, *args, **kwargs):
 473        """
 474        Extend the seismogram.
 475        """
 476        # Merge and trim to pad.
 477        self.stream.merge()
 478        if len(self.stream) != 1:
 479            msg = "All traces need to be of the same id for a dayplot"
 480            raise ValueError(msg)
 481        self.stream.trim(self.starttime, self.endtime, pad=True)
 482        # Get minmax array.
 483        self.__dayplotGetMinMaxValues(self, *args, **kwargs)
 484        # Normalize array
 485        self.__dayplotNormalizeValues(self, *args, **kwargs)
 486        # Get timezone information. If none is given, use local time.
 487        self.time_offset = kwargs.get(
 488            'time_offset',
 489            round((UTCDateTime(datetime.now()) - UTCDateTime()) / 3600.0, 2))
 490        self.timezone = kwargs.get('timezone', 'local time')
 491        # Try to guess how many steps are needed to advance one full time unit.
 492        self.repeat = None
 493        intervals = self.extreme_values.shape[0]
 494        if self.interval < 60 and 60 % self.interval == 0:
 495            self.repeat = 60 // self.interval
 496        elif self.interval < 1800 and 3600 % self.interval == 0:
 497            self.repeat = 3600 // self.interval
 498        # Otherwise use a maximum value of 10.
 499        else:
 500            if intervals >= 10:
 501                self.repeat = 10
 502            else:
 503                self.repeat = intervals
 504        # Create axis to plot on.
 505        if self.background_color:
 506            ax = self.fig.add_subplot(1, 1, 1, axisbg=self.background_color)
 507        else:
 508            ax = self.fig.add_subplot(1, 1, 1)
 509        # Adjust the subplots
 510        self.fig.subplots_adjust(left=self.subplots_adjust_left,
 511                                 right=self.subplots_adjust_right,
 512                                 top=self.subplots_adjust_top,
 513                                 bottom=self.subplots_adjust_bottom)
 514        # Create x_value_array.
 515        x_values = np.repeat(np.arange(self.width), 2)
 516        intervals = self.extreme_values.shape[0]
 517
 518        for _i in range(intervals):
 519            # Create offset array.
 520            y_values = np.ma.empty(self.width * 2)
 521            y_values.fill(intervals - (_i + 1))
 522            # Add min and max values.
 523            y_values[0::2] += self.extreme_values[_i, :, 0]
 524            y_values[1::2] += self.extreme_values[_i, :, 1]
 525            # Plot the values.
 526            ax.plot(x_values, y_values,
 527                    color=self.color[_i % len(self.color)],
 528                    linewidth=self.linewidth, linestyle=self.linestyle)
 529        # Plot the scale, if required.
 530        scale_unit = kwargs.get("data_unit", None)
 531        if scale_unit is not None:
 532            self._plotDayplotScale(unit=scale_unit)
 533        # Set ranges.
 534        ax.set_xlim(0, self.width - 1)
 535        ax.set_ylim(-0.3, intervals + 0.3)
 536        self.axis = [ax]
 537        # Set ticks.
 538        self.__dayplotSetYTicks(*args, **kwargs)
 539        self.__dayplotSetXTicks(*args, **kwargs)
 540        # Choose to show grid but only on the x axis.
 541        self.fig.axes[0].grid(color=self.grid_color,
 542                              linestyle=self.grid_linestyle,
 543                              linewidth=self.grid_linewidth)
 544        self.fig.axes[0].yaxis.grid(False)
 545        # Set the title of the plot.
 546        if self.title is not None:

 567    def _plotEvent(self, event):
 568        """
 569        Helper function to plot an event into the dayplot.
 570        """
 571        ax = self.fig.axes[0]
 572        seed_id = self.stream[0].id
 573        if hasattr(event, "preferred_origin"):
 574            # Get the time from the preferred origin, alternatively the first
 575            # origin.
 576            origin = event.preferred_origin()
 577            if origin is None:
 578                if event.origins:
 579                    origin = event.origins[0]
 580                else:
 581                    return
 582            time = origin.time
 583
 584            # Do the same for the magnitude.
 585            mag = event.preferred_magnitude()
 586            if mag is None:
 587                if event.magnitudes:
 588                    mag = event.magnitudes[0]
 589            if mag is None:
 590                mag = ""
 591            else:
 592                mag = "%.1f %s" % (mag.mag, mag.magnitude_type)
 593
 594            region = FlinnEngdahl().get_region(origin.longitude,
 595                                               origin.latitude)
 596            text = region
 597            if mag:
 598                text += ", %s" % mag
 599        else:
 600            time = event["time"]
 601            text = event["text"] if "text" in event else None
 602
 603        # Nothing to do if the event is not on the plot.
 604        if time < self.starttime or time > self.endtime:
 605            return
 606
 607        # Now find the position of the event in plot coordinates.
 608
 609        def time2xy(time):
 610            frac = (time - self.starttime) / (self.endtime - self.starttime)
 611            int_frac = (self.interval) / (self.endtime - self.starttime)
 612            event_frac = frac / int_frac
 613            y_pos = self.extreme_values.shape[0] - int(event_frac) - 0.5
 614            x_pos = (event_frac - int(event_frac)) * self.width
 615            return x_pos, y_pos
 616
 617        x_pos, y_pos = time2xy(time)
 618
 619        if text:
 620            # Some logic to get a somewhat sane positioning of the annotation
 621            # box and the arrow..
 622            text_offset_x = 0.10 * self.width
 623            text_offset_y = 1.00
 624            # Relpos determines the connection of the arrow on the box in
 625            # relative coordinates.
 626            relpos = [0.0, 0.5]
 627            # Arc strength is the amount of bending of the arrow.
 628            arc_strength = 0.25
 629            if x_pos < (self.width / 2.0):
 630                text_offset_x_sign = 1.0
 631                ha = "left"
 632                # Arc sign determines the direction of bending.
 633                arc_sign = "+"
 634            else:
 635                text_offset_x_sign = -1.0
 636                ha = "right"
 637                relpos[0] = 1.0
 638                arc_sign = "-"
 639            if y_pos < (self.extreme_values.shape[0] / 2.0):
 640                text_offset_y_sign = 1.0
 641                va = "bottom"
 642            else:
 643                text_offset_y_sign = -1.0
 644                va = "top"
 645                if arc_sign == "-":
 646                    arc_sign = "+"
 647                else:
 648                    arc_sign = "-"
 649
 650            # Draw the annotation including box.
 651            ax.annotate(text,
 652                        # The position of the event.
 653                        xy=(x_pos, y_pos),
 654                        # The position of the text, offset depending on the
 655                        # previously calculated variables.
 656                        xytext=(x_pos + text_offset_x_sign * text_offset_x,
 657                                y_pos + text_offset_y_sign * text_offset_y),
 658                        # Everything in data coordinates.
 659                        xycoords="data", textcoords="data",
 660                        # Set the text alignment.
 661                        ha=ha, va=va,
 662                        # Text box style.
 663                        bbox=dict(boxstyle="round", fc="w", alpha=0.6),
 664                        # Arrow style
 665                        arrowprops=dict(
 666                            arrowstyle="-",
 667                            connectionstyle="arc3, rad=%s%.1f" % (
 668                                arc_sign, arc_strength),
 669                            relpos=relpos, shrinkB=7),
 670                        zorder=10)
 671        # Draw the actual point. Use a marker with a star shape.
 672        ax.plot(x_pos, y_pos, "*", color="yellow",
 673                markersize=12, linewidth=self.linewidth)
 674
 675        for pick in getattr(event, 'picks', []):
 676            # check that network/station/location matches
 677            if pick.waveform_id.getSEEDString().split(".")[:-1] != \
 678                    seed_id.split(".")[:-1]:
 679                continue
 680            x_pos, y_pos = time2xy(pick.time)
 681            ax.plot(x_pos, y_pos, "|", color="red",
 682                    markersize=50, markeredgewidth=self.linewidth * 4)

 731        """
 732        Just plots the data samples in the self.stream. Useful for smaller
 733        datasets up to around 1000000 samples (depending on the machine on

 735
 736        Slow and high memory consumption for large datasets.

 739        st = Stream(traces=trace)
 740        self._draw_overlap_axvspans(st, ax)

 742            # Check if it is a preview file and adjust accordingly.
 743            # XXX: Will look weird if the preview file is too small.
 744            if trace.stats.get('preview'):
 745                # Mask the gaps.
 746                trace.data = np.ma.masked_array(trace.data)
 747                trace.data[trace.data == -1] = np.ma.masked
 748                # Recreate the min_max scene.
 749                dtype = trace.data.dtype
 750                old_time_range = trace.stats.endtime - trace.stats.starttime
 751                data = np.empty(2 * trace.stats.npts, dtype=dtype)
 752                data[0::2] = trace.data / 2.0
 753                data[1::2] = -trace.data / 2.0
 754                trace.data = data
 755                # The times are not supposed to change.
 756                trace.stats.delta = (
 757                        old_time_range / float(trace.stats.npts - 1))
 758            trace.data = np.require(trace.data, np.float64) * trace.stats.calib
 759            if self.type == 'relative':
 760                # use seconds of relative sample times and shift by trace's
 761                # start time, which was set relative to `reftime`.
 762                x_values = (

 764            else:
 765                # convert seconds of relative sample times to days and add
 766                # start time of trace.

 769            ax.plot(x_values, trace.data, color=self.color,
 770                    linewidth=self.linewidth, linestyle=self.linestyle)

 772        trace = st[0]

 774            tr_id = trace.id + ' [preview]'
 775        elif hasattr(trace, 'label'):
 776            tr_id = trace.label
 777        else:
 778            tr_id = trace.id
 779        self.ids.append(tr_id)

 781    def __plotMinMax(self, trace, ax, *args, **kwargs):  # @UnusedVariable
 782        """
 783        Plots the data using a min/max approach that calculated the minimum and
 784        maximum values of each "pixel" and then plots only these values. Works
 785        much faster with large data sets.
 786        """
 787        self._draw_overlap_axvspans(Stream(traces=trace), ax)
 788        # Some variables to help calculate the values.
 789        starttime = self._time_to_xvalue(self.starttime)
 790        endtime = self._time_to_xvalue(self.endtime)
 791        # The same trace will always have the same sampling_rate.
 792        sampling_rate = trace[0].stats.sampling_rate
 793        # width of x axis in seconds
 794        x_width = endtime - starttime
 795        # normal plots have x-axis in days, so convert x_width to seconds
 796        if self.type != "relative":
 797            x_width = x_width * SECONDS_PER_DAY
 798        # number of samples that get represented by one min-max pair
 799        pixel_length = int(
 800            np.ceil((x_width * sampling_rate + 1) / self.width))
 801        # Loop over all the traces. Do not merge them as there are many samples
 802        # and therefore merging would be slow.
 803        for _i, tr in enumerate(trace):
 804            trace_length = len(tr.data)
 805            pixel_count = int(trace_length // pixel_length)
 806            remaining_samples = int(trace_length % pixel_length)
 807            remaining_seconds = remaining_samples / sampling_rate
 808            if self.type != "relative":
 809                remaining_seconds /= SECONDS_PER_DAY
 810            # Reference to new data array which does not copy data but can be
 811            # reshaped.
 812            if remaining_samples:
 813                data = tr.data[:-remaining_samples]
 814            else:
 815                data = tr.data
 816            data = data.reshape(pixel_count, pixel_length)
 817            min_ = data.min(axis=1) * tr.stats.calib
 818            max_ = data.max(axis=1) * tr.stats.calib
 819            # Calculate extreme_values and put them into new array.
 820            if remaining_samples:
 821                extreme_values = np.empty((pixel_count + 1, 2), dtype=np.float)
 822                extreme_values[:-1, 0] = min_
 823                extreme_values[:-1, 1] = max_
 824                extreme_values[-1, 0] = \
 825                    tr.data[-remaining_samples:].min() * tr.stats.calib
 826                extreme_values[-1, 1] = \
 827                    tr.data[-remaining_samples:].max() * tr.stats.calib
 828            else:
 829                extreme_values = np.empty((pixel_count, 2), dtype=np.float)
 830                extreme_values[:, 0] = min_
 831                extreme_values[:, 1] = max_
 832            # Finally plot the data.
 833            start = self._time_to_xvalue(tr.stats.starttime)
 834            end = self._time_to_xvalue(tr.stats.endtime)
 835            if remaining_samples:
 836                # the last minmax pair is inconsistent regarding x-spacing
 837                x_values = np.linspace(start, end - remaining_seconds,
 838                                       num=extreme_values.shape[0] - 1)
 839                x_values = np.concatenate([x_values, [end]])
 840            else:
 841                x_values = np.linspace(start, end, num=extreme_values.shape[0])
 842            x_values = np.repeat(x_values, 2)
 843            y_values = extreme_values.flatten()
 844            ax.plot(x_values, y_values, color=self.color)
 845        # remember xlim state and add callback to warn when zooming in
 846        self._initial_xrange = (self._time_to_xvalue(self.endtime) -
 847                                self._time_to_xvalue(self.starttime))
 848        self._minmax_plot_xrange_dangerous = False
 849        ax.callbacks.connect("xlim_changed", self._warn_on_xaxis_zoom)
 850        # set label, write to self.ids
 851        if hasattr(trace[0], 'label'):
 852            tr_id = trace[0].label
 853        else:
 854            tr_id = trace[0].id
 855        self.ids.append(tr_id)

 903    def __dayplotGetMinMaxValues(self, *args, **kwargs):  # @UnusedVariable
 904        """
 905        Takes a Stream object and calculates the min and max values for each
 906        pixel in the dayplot.
 907
 908        Writes a three dimensional array. The first axis is the step, i.e
 909        number of trace, the second is the pixel in that step and the third
 910        contains the minimum and maximum value of the pixel.
 911        """
 912        # Helper variables for easier access.
 913        trace = self.stream[0]
 914        trace_length = len(trace.data)
 915
 916        # Samples per interval.
 917        spi = int(self.interval * trace.stats.sampling_rate)
 918        # Check the approximate number of samples per pixel and raise
 919        # error as fit.
 920        spp = float(spi) / self.width
 921        if spp < 1.0:
 922            msg = """
 923            Too few samples to use dayplot with the given arguments.
 924            Adjust your arguments or use a different plotting method.
 925            """
 926            msg = " ".join(msg.strip().split())
 927            raise ValueError(msg)
 928        # Number of intervals plotted.
 929        noi = float(trace_length) / spi
 930        inoi = int(round(noi))
 931        # Plot an extra interval if at least 2 percent of the last interval
 932        # will actually contain data. Do it this way to lessen floating point
 933        # inaccuracies.
 934        if abs(noi - inoi) > 2E-2:
 935            noi = inoi + 1
 936        else:
 937            noi = inoi
 938
 939        # Adjust data. Fill with masked values in case it is necessary.
 940        number_of_samples = noi * spi
 941        delta = number_of_samples - trace_length
 942        if delta < 0:
 943            trace.data = trace.data[:number_of_samples]
 944        elif delta > 0:
 945            trace.data = np.ma.concatenate(
 946                [trace.data, createEmptyDataChunk(delta, trace.data.dtype)])
 947
 948        # Create array for min/max values. Use masked arrays to handle gaps.
 949        extreme_values = np.ma.empty((noi, self.width, 2))
 950        trace.data.shape = (noi, spi)
 951
 952        ispp = int(spp)
 953        fspp = spp % 1.0
 954        if fspp == 0.0:
 955            delta = None
 956        else:
 957            delta = spi - ispp * self.width
 958
 959        # Loop over each interval to avoid larger errors towards the end.
 960        for _i in range(noi):
 961            if delta:
 962                cur_interval = trace.data[_i][:-delta]
 963                rest = trace.data[_i][-delta:]
 964            else:
 965                cur_interval = trace.data[_i]
 966            cur_interval.shape = (self.width, ispp)
 967            extreme_values[_i, :, 0] = cur_interval.min(axis=1)
 968            extreme_values[_i, :, 1] = cur_interval.max(axis=1)
 969            # Add the rest.
 970            if delta:
 971                extreme_values[_i, -1, 0] = min(extreme_values[_i, -1, 0],
 972                                                rest.min())
 973                extreme_values[_i, -1, 1] = max(extreme_values[_i, -1, 0],
 974                                                rest.max())
 975        # Set class variable.
 976        self.extreme_values = extreme_values

 978    def __dayplotNormalizeValues(self, *args, **kwargs):  # @UnusedVariable
 979        """
 980        Normalizes all values in the 3 dimensional array, so that the minimum
 981        value will be 0 and the maximum value will be 1.
 982
 983        It will also convert all values to floats.
 984        """
 985        # Convert to native floats.
 986        self.extreme_values = self.extreme_values.astype(np.float) * \
 987                              self.stream[0].stats.calib
 988        # Make sure that the mean value is at 0
 989        self.extreme_values -= self.extreme_values.mean()
 990
 991        # Scale so that 99.5 % of the data will fit the given range.
 992        if self.vertical_scaling_range is None:
 993            percentile_delta = 0.005
 994            max_values = self.extreme_values[:, :, 1].compressed()
 995            min_values = self.extreme_values[:, :, 0].compressed()
 996            # Remove masked values.
 997            max_values.sort()
 998            min_values.sort()
 999            length = len(max_values)
1000            index = int((1.0 - percentile_delta) * length)
1001            max_val = max_values[index]
1002            index = int(percentile_delta * length)
1003            min_val = min_values[index]
1004        # Exact fit.
1005        elif float(self.vertical_scaling_range) == 0.0:
1006            max_val = self.extreme_values[:, :, 1].max()
1007            min_val = self.extreme_values[:, :, 0].min()
1008        # Fit with custom range.
1009        else:
1010            max_val = min_val = abs(self.vertical_scaling_range) / 2.0
1011
1012        # Normalization factor.
1013        self._normalization_factor = max(abs(max_val), abs(min_val)) * 2
1014
1015        # Scale from 0 to 1.
1016        self.extreme_values = self.extreme_values / self._normalization_factor
1017        self.extreme_values += 0.5

1191    def __sectInitTraces(self):
1192        """
1193        Arrange the trace data used for plotting.
1194
1195        If necessary the data is resampled before
1196        being collected in a continuous list.
1197        """
1198        # Extract distances from st[].stats.distance
1199        # or from st.[].stats.coordinates.latitude...
1200        self._tr_offsets = np.empty(len(self.stream))
1201        if not self.sect_dist_degree:
1202            # Define offset in km from tr.stats.distance
1203            try:
1204                for _i, tr in enumerate(self.stream):
1205                    self._tr_offsets[_i] = tr.stats.distance
1206            except:
1207                msg = 'Define trace.stats.distance in meters to epicenter'
1208                raise ValueError(msg)
1209        else:
1210            # Define offset as degree from epicenter
1211            try:
1212                for _i, tr in enumerate(self.stream):
1213                    self._tr_offsets[_i] = locations2degrees(
1214                        tr.stats.coordinates.latitude,
1215                        tr.stats.coordinates.longitude,
1216                        self.ev_coord[0], self.ev_coord[1])
1217            except:
1218                msg = 'Define latitude/longitude in trace.stats.' + \
1219                      'coordinates and ev_coord. See documentation.'
1220                raise ValueError(msg)
1221        # Define minimum and maximum offsets
1222        if self.sect_offset_min is None:
1223            self._offset_min = self._tr_offsets.min()
1224        else:
1225            self._offset_min = self.sect_offset_min
1226
1227        if self.sect_offset_max is None:
1228            self._offset_max = self._tr_offsets.max()
1229        else:
1230            self._offset_max = self.sect_offset_max
1231        # Reduce data to indexes within offset_min/max
1232        mask = ((self._tr_offsets >= self._offset_min) &
1233                (self._tr_offsets <= self._offset_max))
1234        self._tr_offsets = self._tr_offsets[mask]
1235        stream = [tr for m, tr in zip(mask, self.stream) if m]
1236        # Normalized offsets for plotting
1237        self._tr_offsets_norm = self._tr_offsets / self._tr_offsets.max()
1238        # Number of traces
1239        self._tr_num = len(self._tr_offsets)
1240        # Arranging trace data in single list
1241        self._tr_data = []
1242        # Maximum counts, npts, starttime and delta of each selected trace
1243        self._tr_starttimes = []
1244        self._tr_max_count = np.empty(self._tr_num)
1245        self._tr_npts = np.empty(self._tr_num)
1246        self._tr_delta = np.empty(self._tr_num)
1247        # TODO dynamic DATA_MAXLENGTH according to dpi
1248        for _i, tr in enumerate(stream):
1249            if len(tr.data) >= self.max_npts:
1250                tmp_data = signal.resample(tr.data, self.max_npts)
1251            else:
1252                tmp_data = tr.data
1253            # Initialising trace stats
1254            self._tr_data.append(tmp_data)
1255            self._tr_starttimes.append(tr.stats.starttime)
1256            self._tr_max_count[_i] = tmp_data.max()
1257            self._tr_npts[_i] = tmp_data.size
1258            self._tr_delta[_i] = (
1259                                         tr.stats.endtime -
1260                                         tr.stats.starttime) / self._tr_npts[
1261                                     _i]
1262        # Init time vectors
1263        self.__sectInitTime()

1272    def __sectInitTime(self):
1273        """
1274        Define the time vector for each trace
1275        """
1276        reftime = self.sect_reftime or min(self._tr_starttimes)
1277        self._tr_times = []
1278        for _tr in range(self._tr_num):
1279            self._tr_times.append(
1280                (np.arange(self._tr_npts[_tr]) +
1281                 (self._tr_starttimes[_tr] - reftime)) * self._tr_delta[_tr])
1282            if self.sect_vred:
1283                self._tr_times[-1] -= self._tr_offsets[_tr] / self.sect_vred
1284
1285        self._time_min = np.concatenate(self._tr_times).min()
1286        self._time_max = np.concatenate(self._tr_times).max()

 346        """
 347        Plot the Traces showing one graph per Trace.
 348
 349        Plots the whole time series for self.max_npts points and less. For more
 350        points it plots minmax values.
 351        """
 352        stream_new = []
 353        # Just remove empty traces.
 354        if not self.automerge:
 355            for tr in self.stream:
 356                stream_new.append([])
 357                if len(tr.data):
 358                    stream_new[-1].append(tr)
 359        else:
 360            # Generate sorted list of traces (no copy)
 361            # Sort order: id, starttime, endtime
 362            ids = self.__getMergablesIds()
 363            for id in ids:
 364                stream_new.append([])
 365                for tr in self.stream:
 366                    tr_id = self.__getMergeId(tr)
 367                    if tr_id == id:
 368                        # does not copy the elements of the data array
 369                        tr_ref = copy(tr)
 370                        if tr_ref.data.size:
 371                            stream_new[-1].append(tr_ref)
 372                # delete if empty list
 373                if not len(stream_new[-1]):
 374                    stream_new.pop()
 375                    continue
 376        # If everything is lost in the process raise an Exception.
 377        if not len(stream_new):
 378            raise Exception("Nothing to plot")
 379        # Create helper variable to track ids and min/max/mean values.
 380        self.ids = []
 381        # Loop over each Trace and call the appropriate plotting method.
 382        self.axis = []
 383
 384        import numpy as np
 385        if self.event and self.location:
 386            origin = self.event.preferred_origin()
 387            if origin:
 388                ex = origin.x
 389                ey = origin.y
 390                ez = origin.z
 391                eloc = np.array([ex, ey, ez])
 392
 393                traces = []
 394                name = []
 395                dist = []
 396
 397                for sta in self.location.stations():
 398                    sloc = np.array([sta.x, sta.y, sta.z])
 399                    dist.append(np.linalg.norm(eloc - sloc))
 400                    name.append(sta.code)
 401
 402                indices = np.argsort(dist)
 403                name = np.array(name)
 404                traces = []
 405                for nm in name[indices]:
 406                    for tr in stream_new:
 407                        if tr[0].stats.station == nm:
 408                            traces.append(tr)
 409
 410                stream_new = traces
 411
 412        for _i, tr in enumerate(stream_new):
 413            # Each trace needs to have the same sampling rate.
 414            sampling_rates = set([_tr.stats.sampling_rate for _tr in tr])
 415            if len(sampling_rates) > 1:
 416                msg = "All traces with the same id need to have the same " + \
 417                      "sampling rate."
 418                raise Exception(msg)
 419            sampling_rate = sampling_rates.pop()
 420            if _i == 0:
 421                sharex = None
 422            else:
 423                sharex = self.axis[0]
 424            ax = self.fig.add_subplot(len(stream_new), 1, _i + 1,
 425                                      # axisbg=self.background_color,
 426                                      sharex=sharex)
 427            self.axis.append(ax)
 428            # XXX: Also enable the minmax plotting for previews.
 429            method_ = self.plotting_method
 430            if method_ is None:
 431                if ((self.endtime - self.starttime) * sampling_rate >
 432                        self.max_npts):
 433                    method_ = "fast"
 434                else:
 435                    method_ = "full"
 436            method_ = method_.lower()
 437            if method_ == 'full':
 438                self.__plotStraight(stream_new[_i], ax, *args, **kwargs)
 439            elif method_ == 'fast':
 440                self.__plotMinMax(stream_new[_i], ax, *args, **kwargs)
 441            else:
 442                msg = "Invalid plot method: '%s'" % method_
 443                raise ValueError(msg)
 444
 445            # plot picks

 449                    pk = arrival.get_pick()
 450                    if (pk.waveform_id.station_code == tr[0].stats.station) \
 451                            & (pk.waveform_id.location_code ==
 452                               tr[0].stats.location):
 453                        time = date2num(pk.time.datetime)
 454                        ax.axvline(time, ls='-',
 455                                   c=colors[pk.phase_hint.upper()])
 456                        if arrival.time_residual is not None:
 457                            # predicted_time = time - arrival.time_residual
 458                            predicted_time = pk.time - \
 459                                             arrival.time_residual
 460                            ax.axvline(date2num(predicted_time),
 461                                       ls='--',
 462                                       c=colors[pk.phase_hint.upper()])
 463                            # raspoutine
 464
 465        # Set ticks.
 466        self.__plotSetXTicks()
 467        self.__plotSetYTicks()
 468        xmin = self._time_to_xvalue(self.starttime)
 469        xmax = self._time_to_xvalue(self.endtime)
 470        ax.set_xlim(xmin, xmax)

 474        """
 475        Extend the seismogram.
 476        """
 477        # Merge and trim to pad.
 478        self.stream.merge()
 479        if len(self.stream) != 1:
 480            msg = "All traces need to be of the same id for a dayplot"
 481            raise ValueError(msg)
 482        self.stream.trim(self.starttime, self.endtime, pad=True)
 483        # Get minmax array.
 484        self.__dayplotGetMinMaxValues(self, *args, **kwargs)
 485        # Normalize array

 489            'time_offset',
 490            round((UTCDateTime(datetime.now()) - UTCDateTime()) / 3600.0, 2))
 491        self.timezone = kwargs.get('timezone', 'local time')
 492        # Try to guess how many steps are needed to advance one full time unit.
 493        self.repeat = None
 494        intervals = self.extreme_values.shape[0]
 495        if self.interval < 60 and 60 % self.interval == 0:
 496            self.repeat = 60 // self.interval
 497        elif self.interval < 1800 and 3600 % self.interval == 0:
 498            self.repeat = 3600 // self.interval
 499        # Otherwise use a maximum value of 10.
 500        else:
 501            if intervals >= 10:
 502                self.repeat = 10
 503            else:
 504                self.repeat = intervals

 507            ax = self.fig.add_subplot(1, 1, 1, axisbg=self.background_color)
 508        else:
 509            ax = self.fig.add_subplot(1, 1, 1)
 510        # Adjust the subplots
 511        self.fig.subplots_adjust(left=self.subplots_adjust_left,
 512                                 right=self.subplots_adjust_right,
 513                                 top=self.subplots_adjust_top,
 514                                 bottom=self.subplots_adjust_bottom)

 516        x_values = np.repeat(np.arange(self.width), 2)
 517        intervals = self.extreme_values.shape[0]

 519        for _i in range(intervals):
 520            # Create offset array.
 521            y_values = np.ma.empty(self.width * 2)
 522            y_values.fill(intervals - (_i + 1))

 536        ax.set_ylim(-0.3, intervals + 0.3)
 537        self.axis = [ax]
 538        # Set ticks.
 539        self.__dayplotSetYTicks(*args, **kwargs)
 540        self.__dayplotSetXTicks(*args, **kwargs)
 541        # Choose to show grid but only on the x axis.
 542        self.fig.axes[0].grid(color=self.grid_color,
 543                              linestyle=self.grid_linestyle,
 544                              linewidth=self.grid_linewidth)
 545        self.fig.axes[0].yaxis.grid(False)
 546        # Set the title of the plot.
 547        if self.title is not None:
 548            self.fig.suptitle(self.title, fontsize=self.title_size)
 549        # Now try to plot some events.
 550        events = kwargs.get("events", [])
 551        # Potentially download some events with the help of obspy.fdsn.

 568    def _plotEvent(self, event):
 569        """
 570        Helper function to plot an event into the dayplot.
 571        """
 572        ax = self.fig.axes[0]
 573        seed_id = self.stream[0].id
 574        if hasattr(event, "preferred_origin"):
 575            # Get the time from the preferred origin, alternatively the first
 576            # origin.
 577            origin = event.preferred_origin()
 578            if origin is None:
 579                if event.origins:
 580                    origin = event.origins[0]
 581                else:
 582                    return
 583            time = origin.time
 584
 585            # Do the same for the magnitude.
 586            mag = event.preferred_magnitude()
 587            if mag is None:
 588                if event.magnitudes:
 589                    mag = event.magnitudes[0]
 590            if mag is None:
 591                mag = ""
 592            else:
 593                mag = "%.1f %s" % (mag.mag, mag.magnitude_type)
 594
 595            region = FlinnEngdahl().get_region(origin.longitude,
 596                                               origin.latitude)
 597            text = region
 598            if mag:
 599                text += ", %s" % mag
 600        else:
 601            time = event["time"]
 602            text = event["text"] if "text" in event else None
 603
 604        # Nothing to do if the event is not on the plot.
 605        if time < self.starttime or time > self.endtime:
 606            return
 607
 608        # Now find the position of the event in plot coordinates.
 609
 610        def time2xy(time):
 611            frac = (time - self.starttime) / (self.endtime - self.starttime)
 612            int_frac = (self.interval) / (self.endtime - self.starttime)
 613            event_frac = frac / int_frac
 614            y_pos = self.extreme_values.shape[0] - int(event_frac) - 0.5
 615            x_pos = (event_frac - int(event_frac)) * self.width
 616            return x_pos, y_pos
 617
 618        x_pos, y_pos = time2xy(time)
 619
 620        if text:
 621            # Some logic to get a somewhat sane positioning of the annotation
 622            # box and the arrow..
 623            text_offset_x = 0.10 * self.width
 624            text_offset_y = 1.00
 625            # Relpos determines the connection of the arrow on the box in
 626            # relative coordinates.
 627            relpos = [0.0, 0.5]
 628            # Arc strength is the amount of bending of the arrow.
 629            arc_strength = 0.25
 630            if x_pos < (self.width / 2.0):
 631                text_offset_x_sign = 1.0
 632                ha = "left"
 633                # Arc sign determines the direction of bending.
 634                arc_sign = "+"
 635            else:
 636                text_offset_x_sign = -1.0
 637                ha = "right"
 638                relpos[0] = 1.0
 639                arc_sign = "-"
 640            if y_pos < (self.extreme_values.shape[0] / 2.0):
 641                text_offset_y_sign = 1.0
 642                va = "bottom"
 643            else:
 644                text_offset_y_sign = -1.0
 645                va = "top"
 646                if arc_sign == "-":
 647                    arc_sign = "+"
 648                else:
 649                    arc_sign = "-"
 650
 651            # Draw the annotation including box.
 652            ax.annotate(text,
 653                        # The position of the event.
 654                        xy=(x_pos, y_pos),
 655                        # The position of the text, offset depending on the
 656                        # previously calculated variables.
 657                        xytext=(x_pos + text_offset_x_sign * text_offset_x,
 658                                y_pos + text_offset_y_sign * text_offset_y),
 659                        # Everything in data coordinates.
 660                        xycoords="data", textcoords="data",
 661                        # Set the text alignment.
 662                        ha=ha, va=va,
 663                        # Text box style.
 664                        bbox=dict(boxstyle="round", fc="w", alpha=0.6),
 665                        # Arrow style
 666                        arrowprops=dict(
 667                            arrowstyle="-",
 668                            connectionstyle="arc3, rad=%s%.1f" % (
 669                                arc_sign, arc_strength),
 670                            relpos=relpos, shrinkB=7),
 671                        zorder=10)
 672        # Draw the actual point. Use a marker with a star shape.
 673        ax.plot(x_pos, y_pos, "*", color="yellow",
 674                markersize=12, linewidth=self.linewidth)
 675
 676        for pick in getattr(event, 'picks', []):
 677            # check that network/station/location matches
 678            if pick.waveform_id.getSEEDString().split(".")[:-1] != \
 679                    seed_id.split(".")[:-1]:
 680                continue
 681            x_pos, y_pos = time2xy(pick.time)
 682            ax.plot(x_pos, y_pos, "|", color="red",
 683                    markersize=50, markeredgewidth=self.linewidth * 4)

 731    def __plotStraight(self, trace, ax, *args, **kwargs):  # @UnusedVariable
 732        """
 733        Just plots the data samples in the self.stream. Useful for smaller
 734        datasets up to around 1000000 samples (depending on the machine on
 735        which it's being run).
 736
 737        Slow and high memory consumption for large datasets.
 738        """
 739        # trace argument seems to actually be a list of traces..
 740        st = Stream(traces=trace)
 741        self._draw_overlap_axvspans(st, ax)
 742        for trace in st:
 743            # Check if it is a preview file and adjust accordingly.
 744            # XXX: Will look weird if the preview file is too small.
 745            if trace.stats.get('preview'):
 746                # Mask the gaps.
 747                trace.data = np.ma.masked_array(trace.data)
 748                trace.data[trace.data == -1] = np.ma.masked
 749                # Recreate the min_max scene.
 750                dtype = trace.data.dtype
 751                old_time_range = trace.stats.endtime - trace.stats.starttime
 752                data = np.empty(2 * trace.stats.npts, dtype=dtype)
 753                data[0::2] = trace.data / 2.0
 754                data[1::2] = -trace.data / 2.0
 755                trace.data = data
 756                # The times are not supposed to change.
 757                trace.stats.delta = (
 758                        old_time_range / float(trace.stats.npts - 1))
 759            trace.data = np.require(trace.data, np.float64) * trace.stats.calib
 760            if self.type == 'relative':
 761                # use seconds of relative sample times and shift by trace's
 762                # start time, which was set relative to `reftime`.
 763                x_values = (
 764                        trace.times() + (trace.stats.starttime - self.reftime))
 765            else:
 766                # convert seconds of relative sample times to days and add
 767                # start time of trace.
 768                x_values = ((trace.times() / SECONDS_PER_DAY) +
 769                            date2num(trace.stats.starttime.datetime))
 770            ax.plot(x_values, trace.data, color=self.color,
 771                    linewidth=self.linewidth, linestyle=self.linestyle)
 772        # Write to self.ids
 773        trace = st[0]
 774        if trace.stats.get('preview'):
 775            tr_id = trace.id + ' [preview]'
 776        elif hasattr(trace, 'label'):
 777            tr_id = trace.label
 778        else:
 779            tr_id = trace.id
 780        self.ids.append(tr_id)

 782    def __plotMinMax(self, trace, ax, *args, **kwargs):  # @UnusedVariable
 783        """
 784        Plots the data using a min/max approach that calculated the minimum and
 785        maximum values of each "pixel" and then plots only these values. Works
 786        much faster with large data sets.
 787        """
 788        self._draw_overlap_axvspans(Stream(traces=trace), ax)
 789        # Some variables to help calculate the values.
 790        starttime = self._time_to_xvalue(self.starttime)
 791        endtime = self._time_to_xvalue(self.endtime)
 792        # The same trace will always have the same sampling_rate.
 793        sampling_rate = trace[0].stats.sampling_rate
 794        # width of x axis in seconds
 795        x_width = endtime - starttime
 796        # normal plots have x-axis in days, so convert x_width to seconds
 797        if self.type != "relative":
 798            x_width = x_width * SECONDS_PER_DAY
 799        # number of samples that get represented by one min-max pair
 800        pixel_length = int(
 801            np.ceil((x_width * sampling_rate + 1) / self.width))
 802        # Loop over all the traces. Do not merge them as there are many samples
 803        # and therefore merging would be slow.
 804        for _i, tr in enumerate(trace):
 805            trace_length = len(tr.data)
 806            pixel_count = int(trace_length // pixel_length)
 807            remaining_samples = int(trace_length % pixel_length)
 808            remaining_seconds = remaining_samples / sampling_rate
 809            if self.type != "relative":
 810                remaining_seconds /= SECONDS_PER_DAY
 811            # Reference to new data array which does not copy data but can be
 812            # reshaped.
 813            if remaining_samples:
 814                data = tr.data[:-remaining_samples]
 815            else:
 816                data = tr.data
 817            data = data.reshape(pixel_count, pixel_length)
 818            min_ = data.min(axis=1) * tr.stats.calib
 819            max_ = data.max(axis=1) * tr.stats.calib
 820            # Calculate extreme_values and put them into new array.
 821            if remaining_samples:
 822                extreme_values = np.empty((pixel_count + 1, 2), dtype=np.float)
 823                extreme_values[:-1, 0] = min_
 824                extreme_values[:-1, 1] = max_
 825                extreme_values[-1, 0] = \
 826                    tr.data[-remaining_samples:].min() * tr.stats.calib
 827                extreme_values[-1, 1] = \
 828                    tr.data[-remaining_samples:].max() * tr.stats.calib
 829            else:
 830                extreme_values = np.empty((pixel_count, 2), dtype=np.float)
 831                extreme_values[:, 0] = min_
 832                extreme_values[:, 1] = max_
 833            # Finally plot the data.
 834            start = self._time_to_xvalue(tr.stats.starttime)
 835            end = self._time_to_xvalue(tr.stats.endtime)
 836            if remaining_samples:
 837                # the last minmax pair is inconsistent regarding x-spacing
 838                x_values = np.linspace(start, end - remaining_seconds,
 839                                       num=extreme_values.shape[0] - 1)
 840                x_values = np.concatenate([x_values, [end]])
 841            else:
 842                x_values = np.linspace(start, end, num=extreme_values.shape[0])
 843            x_values = np.repeat(x_values, 2)
 844            y_values = extreme_values.flatten()
 845            ax.plot(x_values, y_values, color=self.color)
 846        # remember xlim state and add callback to warn when zooming in
 847        self._initial_xrange = (self._time_to_xvalue(self.endtime) -
 848                                self._time_to_xvalue(self.starttime))
 849        self._minmax_plot_xrange_dangerous = False
 850        ax.callbacks.connect("xlim_changed", self._warn_on_xaxis_zoom)
 851        # set label, write to self.ids
 852        if hasattr(trace[0], 'label'):
 853            tr_id = trace[0].label
 854        else:
 855            tr_id = trace[0].id
 856        self.ids.append(tr_id)

 904    def __dayplotGetMinMaxValues(self, *args, **kwargs):  # @UnusedVariable
 905        """
 906        Takes a Stream object and calculates the min and max values for each
 907        pixel in the dayplot.
 908
 909        Writes a three dimensional array. The first axis is the step, i.e
 910        number of trace, the second is the pixel in that step and the third
 911        contains the minimum and maximum value of the pixel.
 912        """
 913        # Helper variables for easier access.
 914        trace = self.stream[0]
 915        trace_length = len(trace.data)
 916
 917        # Samples per interval.
 918        spi = int(self.interval * trace.stats.sampling_rate)
 919        # Check the approximate number of samples per pixel and raise
 920        # error as fit.
 921        spp = float(spi) / self.width
 922        if spp < 1.0:
 923            msg = """
 924            Too few samples to use dayplot with the given arguments.
 925            Adjust your arguments or use a different plotting method.
 926            """
 927            msg = " ".join(msg.strip().split())
 928            raise ValueError(msg)
 929        # Number of intervals plotted.
 930        noi = float(trace_length) / spi
 931        inoi = int(round(noi))
 932        # Plot an extra interval if at least 2 percent of the last interval
 933        # will actually contain data. Do it this way to lessen floating point
 934        # inaccuracies.
 935        if abs(noi - inoi) > 2E-2:
 936            noi = inoi + 1
 937        else:
 938            noi = inoi
 939
 940        # Adjust data. Fill with masked values in case it is necessary.
 941        number_of_samples = noi * spi
 942        delta = number_of_samples - trace_length
 943        if delta < 0:
 944            trace.data = trace.data[:number_of_samples]
 945        elif delta > 0:
 946            trace.data = np.ma.concatenate(
 947                [trace.data, createEmptyDataChunk(delta, trace.data.dtype)])
 948
 949        # Create array for min/max values. Use masked arrays to handle gaps.
 950        extreme_values = np.ma.empty((noi, self.width, 2))
 951        trace.data.shape = (noi, spi)
 952
 953        ispp = int(spp)
 954        fspp = spp % 1.0
 955        if fspp == 0.0:
 956            delta = None
 957        else:
 958            delta = spi - ispp * self.width
 959
 960        # Loop over each interval to avoid larger errors towards the end.
 961        for _i in range(noi):
 962            if delta:
 963                cur_interval = trace.data[_i][:-delta]
 964                rest = trace.data[_i][-delta:]
 965            else:
 966                cur_interval = trace.data[_i]
 967            cur_interval.shape = (self.width, ispp)
 968            extreme_values[_i, :, 0] = cur_interval.min(axis=1)
 969            extreme_values[_i, :, 1] = cur_interval.max(axis=1)
 970            # Add the rest.
 971            if delta:
 972                extreme_values[_i, -1, 0] = min(extreme_values[_i, -1, 0],
 973                                                rest.min())
 974                extreme_values[_i, -1, 1] = max(extreme_values[_i, -1, 0],
 975                                                rest.max())
 976        # Set class variable.
 977        self.extreme_values = extreme_values

 979    def __dayplotNormalizeValues(self, *args, **kwargs):  # @UnusedVariable
 980        """
 981        Normalizes all values in the 3 dimensional array, so that the minimum
 982        value will be 0 and the maximum value will be 1.
 983
 984        It will also convert all values to floats.
 985        """
 986        # Convert to native floats.
 987        self.extreme_values = self.extreme_values.astype(np.float) * \
 988                              self.stream[0].stats.calib
 989        # Make sure that the mean value is at 0
 990        self.extreme_values -= self.extreme_values.mean()
 991
 992        # Scale so that 99.5 % of the data will fit the given range.
 993        if self.vertical_scaling_range is None:
 994            percentile_delta = 0.005
 995            max_values = self.extreme_values[:, :, 1].compressed()
 996            min_values = self.extreme_values[:, :, 0].compressed()
 997            # Remove masked values.
 998            max_values.sort()
 999            min_values.sort()
1000            length = len(max_values)
1001            index = int((1.0 - percentile_delta) * length)
1002            max_val = max_values[index]
1003            index = int(percentile_delta * length)
1004            min_val = min_values[index]
1005        # Exact fit.
1006        elif float(self.vertical_scaling_range) == 0.0:
1007            max_val = self.extreme_values[:, :, 1].max()
1008            min_val = self.extreme_values[:, :, 0].min()
1009        # Fit with custom range.
1010        else:
1011            max_val = min_val = abs(self.vertical_scaling_range) / 2.0
1012
1013        # Normalization factor.
1014        self._normalization_factor = max(abs(max_val), abs(min_val)) * 2
1015
1016        # Scale from 0 to 1.
1017        self.extreme_values = self.extreme_values / self._normalization_factor
1018        self.extreme_values += 0.5

1192    def __sectInitTraces(self):
1193        """
1194        Arrange the trace data used for plotting.
1195
1196        If necessary the data is resampled before
1197        being collected in a continuous list.
1198        """
1199        # Extract distances from st[].stats.distance
1200        # or from st.[].stats.coordinates.latitude...
1201        self._tr_offsets = np.empty(len(self.stream))
1202        if not self.sect_dist_degree:
1203            # Define offset in km from tr.stats.distance
1204            try:
1205                for _i, tr in enumerate(self.stream):
1206                    self._tr_offsets[_i] = tr.stats.distance
1207            except:
1208                msg = 'Define trace.stats.distance in meters to epicenter'
1209                raise ValueError(msg)
1210        else:
1211            # Define offset as degree from epicenter
1212            try:
1213                for _i, tr in enumerate(self.stream):
1214                    self._tr_offsets[_i] = locations2degrees(
1215                        tr.stats.coordinates.latitude,
1216                        tr.stats.coordinates.longitude,
1217                        self.ev_coord[0], self.ev_coord[1])
1218            except:
1219                msg = 'Define latitude/longitude in trace.stats.' + \
1220                      'coordinates and ev_coord. See documentation.'
1221                raise ValueError(msg)
1222        # Define minimum and maximum offsets
1223        if self.sect_offset_min is None:
1224            self._offset_min = self._tr_offsets.min()
1225        else:
1226            self._offset_min = self.sect_offset_min
1227
1228        if self.sect_offset_max is None:
1229            self._offset_max = self._tr_offsets.max()
1230        else:
1231            self._offset_max = self.sect_offset_max
1232        # Reduce data to indexes within offset_min/max
1233        mask = ((self._tr_offsets >= self._offset_min) &
1234                (self._tr_offsets <= self._offset_max))
1235        self._tr_offsets = self._tr_offsets[mask]
1236        stream = [tr for m, tr in zip(mask, self.stream) if m]
1237        # Normalized offsets for plotting
1238        self._tr_offsets_norm = self._tr_offsets / self._tr_offsets.max()
1239        # Number of traces
1240        self._tr_num = len(self._tr_offsets)
1241        # Arranging trace data in single list
1242        self._tr_data = []
1243        # Maximum counts, npts, starttime and delta of each selected trace
1244        self._tr_starttimes = []
1245        self._tr_max_count = np.empty(self._tr_num)
1246        self._tr_npts = np.empty(self._tr_num)
1247        self._tr_delta = np.empty(self._tr_num)
1248        # TODO dynamic DATA_MAXLENGTH according to dpi
1249        for _i, tr in enumerate(stream):
1250            if len(tr.data) >= self.max_npts:
1251                tmp_data = signal.resample(tr.data, self.max_npts)
1252            else:
1253                tmp_data = tr.data
1254            # Initialising trace stats
1255            self._tr_data.append(tmp_data)
1256            self._tr_starttimes.append(tr.stats.starttime)
1257            self._tr_max_count[_i] = tmp_data.max()
1258            self._tr_npts[_i] = tmp_data.size
1259            self._tr_delta[_i] = (
1260                                         tr.stats.endtime -
1261                                         tr.stats.starttime) / self._tr_npts[
1262                                     _i]
1263        # Init time vectors
1264        self.__sectInitTime()
